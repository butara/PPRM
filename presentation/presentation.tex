\begin{frame}
\frametitle{k-means}
    \begin{itemize}
    \item well-known and widely used
    \item fast but produces equi-sized clusters (Voronoi)
    \item examples assigned to the closest cluster, centroids updated, repeat until convergence
    \end{itemize}
\end{frame}
